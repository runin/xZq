							var dateformat = function (date, format) {

							        M: date.getMonth() + 1,

							        h: date.getHours(),
							        m: date.getMinutes(),
							        s: date.getSeconds()
							    };
							    format = format.replace(/(M+|d+|h+|m+|s+)/g, function (v) {
							        return ((v.length > 1 ? "0" : "") + eval('z.' + v.slice(-1))).slice(-2);
							    });
							    return format.replace(/(y+)/g, function (v) {
